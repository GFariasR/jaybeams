135  * - if px is off limits (<= empty_bid price or >= empty_offer price)
136  * - if qty <= 0

150  << " px=" << px << " qty=" << qty;
151  throw jb::feed_error(os.str());

551  auto rel_px = price_to_relative(le->first);
552  top_levels_.at(rel_px) = le->second;
553  } else {
554  // no more good prices to move...
555  break;
556  }
557  }
558  // erase the moved prices..
559  bottom_levels_.erase(bottom_levels_.begin(), le);
560  }

564  * (return empty quote price if none available).
565  */
566  price4_t next_best_price() const {

568  auto rel_inside = price_to_relative(px_inside_);
569  for (int i = rel_inside - 1; i != -1; --i) {
570  if (top_levels_.at(i) != 0) {
571  // got the next one...
572  return relative_to_price(i);
573  }
574  }
575  return empty_quote().first;
576  }
