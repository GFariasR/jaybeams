92  using namespace std::chrono;

105 
106  if (ts - last_report_ts_ > reporting_interval_

109  report_rate(ts, "msec", per_msec_rate_);
110  report_rate(ts, "usec", per_usec_rate_);
111  report_arrival(ts, "arrival ", interarrival_);
