170 void jb::config_object::merge(
171  class_overrides& by_class, YAML::Node node) {
172  if (not node.IsMap()) {
173  return;
174  }
175  // ... iterate over the node, searching for nodes with a key starting
176  // with ':' ...
177  for (auto i : node) {
178  // ... no key, probably a sequence, skip ...
179  if (not i.first) {
180  continue;
181  }
182  // ... found a key, check the format ...
183  std::string key = i.first.as<std::string>();
184  if (key.find(":") != 0) {
185  continue;
186  }
187  // ... try to insert into the map ...
188  auto ins = by_class.emplace(key, i.second);
189  if (ins.second == true) {
190  // ... good insert, nothing left to do ...
191  continue;
192  }
193  // ... okay there was a node for the class in the map already,
194  // need to merge the values ...
195  merge_values(ins.first->second, i.second);
196  }
