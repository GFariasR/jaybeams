99  void estimate_delay(
100  array_type const& a, array_type const& b, sum2_type const& sum2,
101  confidence_type& confidence, argmax_type& argmax) {
102  // Validate the input sizes. For some types of timeseries the
103  // alignment may be different too, but we only use the alignment
104  // when the type of timeseries guarantees to always be aligned.
105  if (a.size() != tmpa_.size() or b.size() != tmpa_.size()) {
106  throw std::invalid_argument(
107  "size mismatch in time_delay_estimator<>::estimate_delay()");
108  }
109  // First we apply the Fourier transform to both inputs ...
110  a2tmpa_.execute(a, tmpa_);
111  b2tmpb_.execute(b, tmpb_);
112  // ... then we compute Conj(A) * B for the transformed inputs ...
113  for (std::size_t i = 0; i != tmpa_.num_elements(); ++i) {
114  tmpa_.data()[i] = std::conj(tmpa_.data()[i]) * (tmpb_.data()[i]);
115  }
116  // ... then we compute the inverse Fourier transform to the result ...
117  tmpa2out_.execute(tmpa_, out_);
118 
119  // ... finally we compute confidence and argmax for the result(s) ...
120  std::size_t k = 0;
121  for (std::size_t i = 0; i != num_timeseries_; ++i) {
122  precision_type max_val = std::numeric_limits<precision_type>::min();
123  std::size_t argmax_val = 0;
124  for (std::size_t j = 0; j != nsamples_; ++j, ++k) {
125  if (max_val < out_.data()[k]) {
126  max_val = out_.data()[k];
127  argmax_val = j;
128  }
129  }
130  if (jb::testing::close_enough(sum2[i], precision_type{0}, 1)) {
131  confidence[i] = std::numeric_limits<precision_type>::max();
132  } else {
133  confidence[i] = max_val / sum2[i];
134  }
135  argmax[i] = argmax_val;
136  }
137  }
